AStar(Graph, Start, Goal)

    CREATE priority queue PQ
    INSERT (f=0, Start)

    WHILE PQ not empty DO
        Node ← REMOVE node with lowest f
        IF Node = Goal THEN
            PRINT "Goal Reached"
            EXIT
        END IF
        FOR each Neighbor DO
            g ← path cost
            f ← g + heuristic
            INSERT Neighbor into PQ
        END FOR
    END WHILE

END AStar
